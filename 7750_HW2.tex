\begin{homeworkProblem}
Denote G="Good card", and P="Penalty card".

\begin{enumerate}
\item As we know , 
\begin{center}
$S_A=\{G,G,G,P,P\}$.
\end{center}
Therefore, 
\begin{center}
 $P(\mathit{A} \ Good)=\frac{P(\{G,G,G\})}{P(S_A)}= \frac{3}{5}$
\end{center}
\item
For the reason that player $\mathit{B}$ chooses a card after player $\mathit{A}$. the exhaustive events of $\mathit{B}$ is:
\begin{center}
$S_B=\{G,G,P,P\}$
\end{center}
Therefore, the probability of "B Good" with given "A good" is:
\begin{center}
$P(\mathit{B} \ good| \mathit{A} \ good)=\frac{2}{4}=\frac{1}{2}$
\end{center}
\item
Because $\mathit{A}$ is "bad", $\mathit{B}$'s exhaustive events are:
\begin{center}
$S_B=\{G,G,G,P\}$
\end{center}
Therefore, the "B bad" possibility with given "A bad" is:
\begin{center}
$P(\mathit{B} \ good | \mathit{A} \ bad)=\frac{P(\{G,G,G\})}{P(\{G,G,G,P\})}=\frac{3}{4}$
\end{center}
\item
Based on (1.5.5),
\begin{center}
$P(\mathit{B} \ good \cap \mathit{A} \ good)=P(\mathit{A}\ good)P(\mathit{B} \ good | \mathit{A} \ good)=\frac{3}{5}*\frac{1}{2}=\frac{3}{10}$
\end{center}
\item
Assuming the cards are distinct, we denote the good cards are $\mathit{G}_1$, $\mathit{G}_2$ and $\mathit{G}_3$. And the bad cards are $\mathit{P}_1$ and $\mathit{P}_2$. Therefore, the the sample space of ordered pairs is:
\begin{center}
$S =
\{G_1G_2,G_1G_3,G_1P_1,G_1P_2,$ $G_2G_1,G_2G_3,G_2P_1,G_2P_2,$\\
$G_3G_1,G_3G_2,G_3P_1,G_3P_2,$
$P_1G_1,P_1G_2,P_1G_3,P_1P_2,$
$P_2G_1,P_2G_2,P_2G_3,P_2G_2,\}$
$S_{(B \ good | A \ good)}=\{(G_1,G_2)| G_3\}\ or \{(G_2,G_3)| G_1\} \ or \{(G_1,G_3)|G_2 \}$
$S_{(B \ good \cap A \ good)}=\{G_1G_2,G_1G_3,G_2G_1,G_2G_3,G_3G_1,G_3G_2\}$

\end{center}
Therefore, 
\begin{center}
$P(\mathit{B} \ good \cap \mathit{A} \ good)=\frac{S_{(B \ good | A \ good)}}{S}=\frac{6}{20}=\frac{3}{10}$\\
$P(B \ good | A \ good)=\frac{P(B \ good | A \ good)}{S}=\frac{3}{4}$
\end{center}
\item
Because 
\begin{center}
$P(B \ good)=P(B \ good \cap A \ good)+P(B \ good \cap A \ bad)=\frac{3}{10}+\frac{3}{10}=\frac{3}{5}$
\end{center}
\item
As we know,
\begin{center}
$P(B\ good)P(A\ good | B\ good)=P(A\ good)P(B\ good | A\ good)$
\end{center}
Therefore,
\begin{center}
$P(A\ good | B\ good)=\frac{P(A\ good)P(B\ good | A\ good)}{P(B \ good)}=  \frac{\frac{3}{5}*\frac{1}{2}}{\frac{3}{5}}=\frac{1}{2}$
\end{center}
\end{enumerate}
\end{homeworkProblem}
